# Pinned to the versions the model was verified against. scikit-learn in
# particular is pinned because pickled RandomForest models can fail to load
# across major library versions.
streamlit>=1.37,<2
streamlit-webrtc>=0.47,<0.63
# MediaPipe depends on the NON-headless opencv-contrib-python, whose cv2 links
# libGL.so.1 and libgthread-2.0.so.0 -- system libs that Streamlit Community
# Cloud's current (broken) base image can't install (libglib2.0-0 is
# uninstallable there). We install the HEADLESS build of the same contrib
# package so the cv2 that ends up imported needs no system OpenGL/glib at all.
# It must appear BEFORE mediapipe so it is the cv2 that wins on disk.
opencv-contrib-python-headless>=4.8,<5
# Pinned exactly: on Streamlit Cloud a loose range let pip resolve a mediapipe
# 0.10.x against a protobuf that its C++ bindings reject, so `mp.solutions`
# failed to import and raised "module 'mediapipe' has no attribute 'solutions'".
# 0.10.14 pairs with protobuf 4.25.x (pinned below), which Streamlit 1.50 also
# accepts, giving one consistent set on both local and Cloud.
mediapipe==0.10.14
# Must satisfy BOTH mediapipe 0.10.14 (needs 4.25.3+) and streamlit (<7); this
# range is the overlap and prevents the broken protobuf that killed mp.solutions.
protobuf>=4.25.3,<5
numpy>=1.24,<2
scikit-learn==1.2.0
# Pulled in by streamlit-webrtc; pinned explicitly since it is imported directly.
av>=10,<13
# tornado is a transitive dep of both streamlit and streamlit-webrtc, but the
# Streamlit Cloud image has resolved environments without it -- streamlit-webrtc
# imports it at module load, so declare it explicitly to guarantee it is present.
tornado>=6.1
